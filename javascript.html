5. Bucles

Aprende a repetir acciones con for y while.

Editor JavaScript
// Bucles en JavaScript // Bucle for console.log("Bucle for:"); for (let i = 0; i < 5; i++) { console.log(`Valor: ${i}`); } // Bucle while console.log("\nBucle while:"); let contador = 0; while (contador < 3) { console.log(`Contador: ${contador}`); contador++; } // Iterar sobre un array console.log("\nIterando sobre un array:"); let frutas = ["manzana", "banana", "cereza"]; for (let fruta of frutas) { console.log(fruta.toUpperCase()); }

Editor JavaScript
// Calculadora básica function calculadora() { console.log("1. Suma"); console.log("2. Resta"); console.log("3. Multiplicación"); console.log("4. División"); let opcion = prompt("Elige una opción (1-4): "); let num1 = parseFloat(prompt("Ingresa el primer número: ")); let num2 = parseFloat(prompt("Ingresa el segundo número: ")); if (opcion === '1') { console.log(`Resultado: ${num1 + num2}`); } else if (opcion === '2') { console.log(`Resultado: ${num1 - num2}`); } else if (opcion === '3') { console.log(`Resultado: ${num1 * num2}`); } else if (opcion === '4') { if (num2 !== 0) { console.log(`Resultado: ${num1 / num2}`); } else { console.log("Error: División por cero"); } } else { console.log("Opción inválida"); } } // Para probar la calculadora, descomenta la siguiente línea: // calculadora();
