4. Clases

Programación orientada a objetos en TypeScript.

Editor TypeScript
// Clases en TypeScript class Persona { // Propiedades tipadas nombre: string; edad: number; // Constructor constructor(nombre: string, edad: number) { this.nombre = nombre; this.edad = edad; } // Método saludar(): void { console.log(`Hola, soy ${this.nombre} y tengo ${this.edad} años`); } } // Herencia class Estudiante extends Persona { curso: string; constructor(nombre: string, edad: number, curso: string) { super(nombre, edad); this.curso = curso; } estudiar(): void { console.log(`${this.nombre} está estudiando ${this.curso}`); } } const estudiante = new Estudiante("Ana", 20, "TypeScript"); estudiante.saludar(); estudiante.estudiar();

API con Tipos

Simula una API con tipos estrictos.

Editor TypeScript
// Proyecto: API con TypeScript type Usuario = { id: number; nombre: string; email: string; }; type Producto = { id: number; nombre: string; precio: number; }; class FakeAPI { private usuarios: Usuario[] = [ { id: 1, nombre: "Ana", email: "ana@ejemplo.com" }, { id: 2, nombre: "Carlos", email: "carlos@ejemplo.com" } ]; private productos: Producto[] = [ { id: 1, nombre: "Laptop", precio: 999 }, { id: 2, nombre: "Teléfono", precio: 599 } ]; async getUsuarios(): Promise { return this.usuarios; } async getProductos(): Promise { return this.productos; } async getUsuarioById(id: number): Promise { return this.usuarios.find(u => u.id === id); } } (async () => { const api = new FakeAPI(); const usuarios = await api.getUsuarios(); const productos = await api.getProductos(); const usuario = await api.getUsuarioById(1); console.log("Usuarios:", usuarios); console.log("Productos:", productos); console.log("Usuario con ID 1:", usuario); })();
